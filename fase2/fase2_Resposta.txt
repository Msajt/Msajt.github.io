/* NAO APAGUE AS FUNÇÕES verifica E next */    
function verifica(sprite){
var botao;
	if(sprite.position.x > 400 && sprite.position.y > 400){
  		botao = createButton('Próxima Fase');
		botao.mousePressed(next);
  	}
}

function next(){
	window.open('fase3/editor.html');
}

/* COMECE A PROGRAMAR APÓS ESSA LINHA */      

var sprite;  

function setup(){
 	createCanvas(400,400);
 	sprite = createSprite(10,10,30,30);
}
                 
function draw(){
 	verifica();            
}
